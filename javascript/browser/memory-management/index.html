еҶ…еӯҳз®ЎзҗҶ

JavaScript зҡ„еҶ…еӯҳз”ұеј•ж“ҺиҮӘеҠЁеҲҶй…Қе’Ңеӣһж”¶,ејҖеҸ‘иҖ…дёҚз”ЁжүӢеҠЁ malloc / freeгҖӮдҪҶгҖҢиҮӘеҠЁгҖҚдёҚзӯүдәҺгҖҢдёҚз”Ёз®ЎгҖҚвҖ”вҖ”зҗҶи§ЈеҖјжҖҺд№ҲеӯҳгҖҒжҙ»еӨҡд№…гҖҒдёәд»Җд№ҲжјҸ,жүҚиғҪеҶҷеҮәдёҚеҚЎйЎҝгҖҒдёҚжі„жјҸзҡ„д»Јз ҒгҖӮ

еҶ…еӯҳз®ЎзҗҶиҜҙеҲ°еә•е°ұдёүд»¶дәӢ:еҖјеӯҳеңЁе“Ә (ж Ҳ / е Ҷ)гҖҒжҙ»еӨҡд№… (з”ҹе‘Ҫе‘Ёжңҹ)гҖҒдёәд»Җд№ҲиҜҘжӯ»жІЎжӯ» (жі„жјҸ)гҖӮGC зҡ„е…·дҪ“з®—жі•и§Ғ еһғеңҫеӣһж”¶жңәеҲ¶,жң¬зҜҮи®ІеүҚдёӨд»¶ + жі„жјҸжҺ’жҹҘгҖӮ

ж ҲдёҺе Ҷ:еҖјеӯҳеңЁе“ӘвҖӢ

JS зҡ„еҖјжҢүзұ»еһӢеҲҶдёӨеӨ„еӯҳж”ҫ:

  • еҹәжң¬зұ»еһӢ (stringгҖҒnumberгҖҒbooleanгҖҒnullгҖҒundefinedгҖҒsymbolгҖҒbigint):зӣҙжҺҘеӯҳеңЁж ҲйҮҢ,еӨ§е°Ҹеӣәе®ҡгҖӮ
  • еј•з”Ёзұ»еһӢ (objectгҖҒarrayгҖҒfunction):ж•°жҚ®жң¬дҪ“еӯҳеңЁе ҶйҮҢ,ж ҲйҮҢеҸӘж”ҫдёҖдёӘжҢҮеҗ‘е®ғзҡ„жҢҮй’ҲгҖӮ

еҪўиұЎдҫӢеӯҗ:ж ҲеғҸй…’еә—еүҚеҸ°зҡ„й’ҘеҢҷжҹңвҖ”вҖ”ж јеӯҗе°ҸгҖҒж•°йҮҸеӣәе®ҡ,е°ҸдёңиҘҝ (еҹәжң¬зұ»еһӢ) зӣҙжҺҘеЎһж јеӯҗйҮҢ,еӨ§д»¶е°ұж”ҫдёҖеј гҖҢжҲҝеҚЎгҖҚ(жҢҮй’Ҳ);е ҶеғҸй…’еә—е®ўжҲҝвҖ”вҖ”з©әй—ҙеӨ§,еӨ§д»¶иЎҢжқҺ (еҜ№иұЎ) ж”ҫжҲҝй—ҙйҮҢ,еүҚеҸ°еҸӘж”ҘзқҖжҲҝеҚЎгҖӮ

иөӢеҖјиҜӯд№ү:жӢ·еҖјиҝҳжҳҜжӢ·жҢҮй’ҲвҖӢ

ж Ҳ/е Ҷзҡ„еҢәеҲ«,жңҖзӣҙжҺҘзҡ„еҗҺжһңе°ұжҳҜиөӢеҖјиЎҢдёәдёҚеҗҢ:

// еҹәжң¬зұ»еһӢ:жӢ·зҡ„жҳҜеҖј,дёӨдёӘеҸҳйҮҸеҗ„еӯҳдёҖд»Ҫ,дә’дёҚеҪұе“Қ
let a = 10;
let b = a; // ж ҲйҮҢж–°ејҖдёҖж ј,жҠҠ 10 еӨҚеҲ¶иҝӣеҺ»
b = 20;
console.log(a); // 10вҖ”вҖ”a зә№дёқдёҚеҠЁ

// еј•з”Ёзұ»еһӢ:жӢ·зҡ„жҳҜжҢҮй’Ҳ,дёӨдёӘеҸҳйҮҸжҢҮеҗ‘еҗҢдёҖдёӘе ҶеҜ№иұЎ
let obj1 = { n: 1 };
let obj2 = obj1; // еӨҚеҲ¶зҡ„жҳҜжҢҮй’Ҳ,дёҚжҳҜеҜ№иұЎ
obj2.n = 99;
console.log(obj1.n); // 99вҖ”вҖ”ж”№зҡ„жҳҜеҗҢдёҖдёӘеҜ№иұЎ

еҪўиұЎдҫӢеӯҗ:еҹәжң¬зұ»еһӢиөӢеҖјеғҸеӨҚеҚ°дёҖеј зәё,еҗ„жӢҝеҗ„зҡ„,дҪ ж¶Ӯж”№дҪ зҡ„дёҚеҪұе“ҚжҲ‘;еј•з”Ёзұ»еһӢиөӢеҖјеғҸеӨҡй…ҚдёҖжҠҠеҗҢдёҖдёӘжҲҝй—ҙзҡ„й’ҘеҢҷ,и°ҒиҝӣеҺ»жҗ¬еҠЁе®¶е…·,еҸҰдёҖдёӘдәәжҺЁй—ЁзңӢеҲ°зҡ„д№ҹжҳҜжҗ¬еҠЁеҗҺзҡ„ж ·еӯҗгҖӮ

жҸҗзӨә

иҝҷжӯЈжҳҜгҖҢж·ұжӢ·иҙқ vs жө…жӢ·иҙқгҖҚзҡ„ж №жәҗ:жө…жӢ·иҙқеҸӘеӨҚеҲ¶ж ҲйҮҢзҡ„жҢҮй’Ҳ,дёӨиҖ…д»Қе…ұдә«еҗҢдёҖдёӘе ҶеҜ№иұЎ;ж·ұжӢ·иҙқжүҚдјҡеңЁе ҶйҮҢзңҹжӯЈеӨҚеҲ¶дёҖд»Ҫж–°еҜ№иұЎ,д»ҺжӯӨдә’дёҚзӣёе№ІгҖӮ

еҶ…еӯҳз”ҹе‘Ҫе‘ЁжңҹвҖӢ

д»»дҪ•иҜӯиЁҖзҡ„еҶ…еӯҳйғҪиө°дёүжӯҘ,JS еҸӘжҳҜжҠҠжңҖеҗҺдёҖжӯҘиҮӘеҠЁеҢ–дәҶ:

  1. еҲҶй…Қ:еЈ°жҳҺеҸҳйҮҸгҖҒеҲӣе»әеҜ№иұЎ/ж•°з»„/еҮҪж•°гҖҒжӢјжҺҘеӯ—з¬ҰдёІж—¶,еј•ж“ҺиҮӘеҠЁеҲҶй…ҚеҶ…еӯҳгҖӮ
  2. дҪҝз”Ё:иҜ»еҶҷиҝҷеқ—еҶ…еӯҳгҖӮ
  3. йҮҠж”ҫ:дёҚеҶҚйңҖиҰҒж—¶з”ұ GC иҮӘеҠЁеӣһж”¶,дҫқжҚ®жҳҜгҖҢеҸҜиҫҫжҖ§гҖҚвҖ”вҖ”д»Һж №иҝҳиғҪи®ҝй—®еҲ°зҡ„е°ұз•ҷзқҖ,и®ҝй—®дёҚеҲ°зҡ„жүҚеӣһж”¶ (иҜҰи§Ғ еһғеңҫеӣһж”¶жңәеҲ¶)гҖӮ

е…ій”®и®ӨзҹҘ:йҮҠж”ҫзҡ„ж—¶жңәдҪ жҺ§еҲ¶дёҚдәҶ,дҪҶиғҪдёҚиғҪйҮҠж”ҫдҪ еҶіе®ҡеҫ—дәҶгҖӮеҸӘиҰҒиҝҳжңүеј•з”ЁжҢҮзқҖдёҖеқ—еҶ…еӯҳ,GC е°ұдёҚж•ўеӣһж”¶вҖ”вҖ”иҝҷе°ұеј•еҮәдәҶеҶ…еӯҳжі„жјҸгҖӮ

еҶ…еӯҳжі„жјҸ:иҜҘйҮҠж”ҫеҚҙжІЎйҮҠж”ҫвҖӢ

еҶ…еӯҳжі„жјҸ = йҖ»иҫ‘дёҠе·Із»Ҹз”ЁдёҚеҲ°гҖҒдҪҶд»Қз„¶еҸҜиҫҫ,еҜјиҮҙ GC дёҚиғҪеӣһж”¶зҡ„еҶ…еӯҳгҖӮжјҸеҫ—еӨҡдәҶ,йЎөйқўи¶Ҡз”Ёи¶ҠеҚЎ,жңҖз»Ҳеҙ©жәғгҖӮеӣӣдёӘз»Ҹе…ёеңәжҷҜ:

1. ж„ҸеӨ–зҡ„е…ЁеұҖеҸҳйҮҸвҖӢ

function foo() {
  bar = 'oops'; // еҝҳдәҶ let/const,bar иҮӘеҠЁжҢӮеҲ° window дёҠ,жҲҗдәҶе…ЁеұҖеҸҳйҮҸ
}

е…ЁеұҖеҸҳйҮҸжҢӮеңЁж №еҜ№иұЎ (window) дёҠ,ж•ҙйЎөз”ҹе‘Ҫе‘ЁжңҹйғҪеҸҜиҫҫ,д»ҺеӨҙеҲ°е°ҫдёҚеӣһж”¶гҖӮејҖеҗҜ 'use strict' дёҘж јжЁЎејҸиғҪи®©иҝҷз§ҚжјҸеҶҷзӣҙжҺҘжҠӣй”ҷ,д»ҺжәҗеӨҙе өдҪҸгҖӮ

2. иў«йҒ—еҝҳзҡ„е®ҡж—¶еҷЁ / еӣһи°ғвҖӢ

const data = getBigData(); // дёҖеӨ§еқ—ж•°жҚ®
setInterval(() => {
  doSomething(data); // еӣһи°ғй—ӯеҢ…дёҖзӣҙеј•з”Ё data
}, 1000);
// з»„д»¶еҚёиҪҪ / йЎөйқўеҲҮиө°ж—¶еҝҳдәҶ clearInterval вҶ’ е®ҡж—¶еҷЁдёҚеҒң,data ж°ёиҝңжҙ»зқҖ

еҜ№зӯ–еҫҲз®ҖеҚ•:и°ҒејҖзҡ„е®ҡж—¶еҷЁгҖҒзӣ‘еҗ¬,и°ҒиҙҹиҙЈеңЁдёҚз”Ёж—¶ clearInterval / removeEventListenerгҖӮ

3. жёёзҰ»зҡ„ DOM еј•з”ЁвҖӢ

let btn = document.getElementById('btn');
document.body.removeChild(btn); // д»ҺйЎөйқўз§»йҷӨдәҶиҝҷдёӘиҠӮзӮ№
// дҪҶ JS йҮҢ btn еҸҳйҮҸиҝҳжҢҮзқҖе®ғ,иҝҷдёӘ DOM иҠӮзӮ№еӣһж”¶дёҚжҺү
btn = null; // жүӢеҠЁж–ӯејҖеј•з”Ё,жүҚиғҪи®©е®ғиў«еӣһж”¶

еҪўиұЎдҫӢеӯҗ:дҪ жҠҠе®¶е…·жҗ¬еҮәдәҶжҲҝй—ҙ (д»Һ DOM ж ‘з§»йҷӨ),дҪҶжүӢйҮҢиҝҳж”ҘзқҖе®ғзҡ„йҒҘжҺ§еҷЁ (JS еј•з”Ё),жё…жҙҒе·Ҙ (GC) зңӢдҪ иҝҳжӢҝзқҖ,е°ұдёҚж•ўеҪ“еһғеңҫжү”жҺүгҖӮ

4. дёҚеҪ“зҡ„й—ӯеҢ…вҖӢ

function outer() {
  const huge = new Array(1000000).fill('*'); // еҚ дёҖеӨ§еқ—еҶ…еӯҳ
  return function inner() {
    console.log(huge.length); // inner й—ӯеҢ…еј•з”ЁдәҶ huge
  };
}
const fn = outer(); // еҸӘиҰҒ fn иҝҳеңЁ,huge е°ұиў«й—ӯеҢ…жӢҙдҪҸ,еӣһж”¶дёҚдәҶ

й—ӯеҢ…жң¬иә«дёҚжҳҜжі„жјҸ,й•ҝжңҹжҢҒжңүдёҚеҶҚйңҖиҰҒзҡ„еӨ§еҜ№иұЎжүҚжҳҜгҖӮз”Ёе®ҢжҠҠ fn = null ж–ӯејҖеҚіеҸҜгҖӮй—ӯеҢ…зҡ„еҺҹзҗҶи§Ғ дҪңз”ЁеҹҹдёҺй—ӯеҢ…гҖӮ

з”Ё DevTools жҺ’жҹҘвҖӢ

е®ҡдҪҚжі„жјҸйқ е·Ҙе…·,дёҚйқ зҢң:

  • Memory йқўжқҝ вҶ’ Heap Snapshot (е Ҷеҝ«з…§):еңЁж“ҚдҪңеүҚеҗҺеҗ„жӢҚдёҖеј еҝ«з…§,еҜ№жҜ”зңӢе“ӘдәӣеҜ№иұЎж•°йҮҸеҸӘеўһдёҚеҮҸгҖҒиҝҹиҝҹдёҚеӣһж”¶гҖӮ
  • Performance йқўжқҝ:еҪ•дёҖж®өеҸҚеӨҚж“ҚдҪң,зңӢ JS Heap жӣІзәҝгҖӮеҒҘеә·зҡ„жӣІзәҝжҳҜй”ҜйҪҝзҠ¶ (еҲҶй…ҚвҶ’GC жӢүеӣһвҶ’еҶҚеҲҶй…Қ);еҰӮжһңжҳҜеҸӘеҚҮдёҚйҷҚзҡ„еҸ°йҳ¶ејҸзҲ¬еқЎ,е°ұжҳҜжі„жјҸдҝЎеҸ·гҖӮ

еҪўиұЎдҫӢеӯҗ:еҶ…еӯҳжӣІзәҝеғҸеҝғз”өеӣҫвҖ”вҖ”и§„еҫӢзҡ„й”ҜйҪҝ (ж¶ЁдәҶеҸҲиў«еӣһж”¶) жҳҜеҒҘеә·зҡ„,дёҖи·ҜеҸӘзҲ¬дёҚйҷҚзҡ„еқЎе°ұжҳҜз—…гҖӮ
